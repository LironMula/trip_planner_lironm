  1. צור מאגר ב-GitHub (למשל trip-planner)
  2. צור Fine-grained token עם הרשאת Contents: Read and Write

CREATE TABLE trip_chat (
  id BIGSERIAL PRIMARY KEY,
  proj_key TEXT, plan_key TEXT,
  username TEXT, message TEXT,
  created_at TIMESTAMPTZ DEFAULT now()

ALTER TABLE trip_chat ENABLE ROW LEVEL SECURITY;
CREATE POLICY "all" ON trip_chat FOR ALL USING (true) WITH CHECK (true);
